%%% Prostředí pro důkazy

\renewenvironment{proof}[1][]{
  \par\medskip\noindent
  \textit{\ifthenelse{\equal{#1}{}}
  {Důkaz}
  {#1}}.
}{
\hspace*{\fill}$\qedsymbol$\par\medskip
}
\newenvironment{solution}[1][]{
  \par\medskip\noindent
  \textit{\ifthenelse{\equal{#1}{}}
  {Řešení}
  {#1}}.
}{
\hspace*{\fill}$\qedsymbol$\par\medskip
}
